pipeline {
   agent any
   tools {
      terraform "terraform"
   }

   stages {
      stage('getscm') {
         steps {
git credentialsId: 'github_credentials', url: 'https://github.com/jmstechhome8/create_ec2.git'
          
         }
      }
         
         stage('terraforminit'){
             
           steps{
                   sh "terraform init"
             }
         }
         
         stage('deploy') {
             steps{
                    sh "terraform apply"
             }
             }
         }
   }
